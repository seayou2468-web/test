   132	- (void)cleanupInternal {
   133	    [self log:@"[CLEANUP] Invalidating timers and freeing handles..."];
   134	    dispatch_async(dispatch_get_main_queue(), ^{ [[UIApplication sharedApplication] setIdleTimerDisabled:NO]; });
   135	    if (self.heartbeatTimer) { [self.heartbeatTimer invalidate]; self.heartbeatTimer = nil; }
   136	    if (self.keepAliveTimer) { [self.keepAliveTimer invalidate]; self.keepAliveTimer = nil; }
   137
   138	    if (_springboard) { [self log:[NSString stringWithFormat:@"[CLEANUP] Freeing Springboard (%p)", _springboard]]; springboard_services_free(_springboard); _springboard = NULL; }
   139	    if (_instproxy) { [self log:[NSString stringWithFormat:@"[CLEANUP] Freeing InstProxy (%p)", _instproxy]]; installation_proxy_client_free(_instproxy); _instproxy = NULL; }
   140	    if (_heartbeat) { [self log:[NSString stringWithFormat:@"[CLEANUP] Freeing Heartbeat (%p)", _heartbeat]]; heartbeat_client_free(_heartbeat); _heartbeat = NULL; }
   141	    if (_lockdown) { [self log:[NSString stringWithFormat:@"[CLEANUP] Freeing Lockdown (%p)", _lockdown]]; lockdownd_client_free(_lockdown); _lockdown = NULL; }
   142	    if (_provider) { [self log:[NSString stringWithFormat:@"[CLEANUP] Freeing Provider (%p)", _provider]]; idevice_provider_free(_provider); _provider = NULL; }
   143	    if (_pairingFile) { [self log:[NSString stringWithFormat:@"[CLEANUP] Freeing PairingFile (%p)", _pairingFile]]; idevice_pairing_file_free(_pairingFile); _pairingFile = NULL; }
   144
   145	    [self updateStatus:@"Released" color:[UIColor blackColor]];
   146	    dispatch_async(dispatch_get_main_queue(), ^{
   147	        [self.connectButton setEnabled:YES];
   148	        [self.disconnectButton setEnabled:NO];
   149	    });
   150	    [self log:@"[CLEANUP] All handles released."];
   151	}

   163	- (void)documentPicker:(UIDocumentPickerViewController *)controller didPickDocumentsAtURLs:(NSArray<NSURL *> *)urls {
   164	    NSURL *url = [urls firstObject];
   165	    if (!url) return;
   166
   167	    [self log:[NSString stringWithFormat:@"[PICKER] Selected: %@", [url lastPathComponent]]];
   168	    [self updateStatus:@"Loading Data..." color:[UIColor orangeColor]];
   169
   170	    dispatch_async(_connectionQueue, ^{
   171	        BOOL canAccess = [url startAccessingSecurityScopedResource];
   172	        NSError *error = nil;
   173	        NSData *data = [NSData dataWithContentsOfURL:url options:0 error:&error];
   174	        if (canAccess) [url stopAccessingSecurityScopedResource];
   175
   176	        if (!data) {
   177	            [self log:[NSString stringWithFormat:@"[ERROR] NSData READ FAILED: %@", error.localizedDescription]];
   178	            [self updateStatus:@"Read Error" color:[UIColor redColor]];
   179	            return;
   180	        }
   181
   182	        [self log:[NSString stringWithFormat:@"[PICKER] Data loaded into memory (%lu bytes).", (unsigned long)data.length]];
   183	        if (data.length > 8) {
   184	            char header[9];
   185	            [data getBytes:header length:8];
   186	            header[8] = '\0';
   187	            [self log:[NSString stringWithFormat:@"[PICKER] File Header Preview: %s", header]];
   188	        }
   189
   190	        dispatch_async(dispatch_get_main_queue(), ^{
   191	            [self.connectButton setEnabled:NO];
   192	            [self.disconnectButton setEnabled:YES];
   193	        });
   194
   195	        self->_activeToken++;
   196	        [self performConnectWithData:data withToken:self->_activeToken];
   197	    });
   198

   200	// FIX: Renamed method to match call site 'performConnectWithData:withToken:'
   201	- (void)performConnectWithData:(NSData *)data withToken:(NSInteger)token {
   202	    [self log:[NSString stringWithFormat:@"[CONN] Starting sequence (Token: %ld)", (long)token]];
   203	    [self cleanupInternal];
   204
   205	    auto check = ^BOOL() { return (self->_activeToken == token); };
   206
   207	    if (!check()) return;
   208	    [self log:@"[STEP 1] Calling idevice_pairing_file_from_bytes..."];
   209	    struct IdeviceFfiError *err = idevice_pairing_file_from_bytes((const uint8_t *)data.bytes, (uintptr_t)data.length, &_pairingFile);
   210	    if (err || !_pairingFile) {
   211	        [self log:[NSString stringWithFormat:@"[ERROR] Parsing pairing data: %s (%d)", err ? err->message : "NULL_ERR", err ? err->code : -1]];
   212	        if (err) { idevice_error_free(err); err = NULL; }
   213	        [self cleanupInternal];
   214	        return;
   215	    }
   216	    [self log:[NSString stringWithFormat:@"[OK] PairingFile parsed (%p).", _pairingFile]];
   217	    if (err) { idevice_error_free(err); err = NULL; }
   218
   219	    if (!check()) return;
   220	    [self log:@"[STEP 2] Setting up sockaddr for 10.7.0.1:62078..."];
   221	    struct sockaddr_in addr;
   222	    memset(&addr, 0, sizeof(addr));
   223	    addr.sin_len = sizeof(addr);
   224	    addr.sin_family = AF_INET;
   225	    addr.sin_port = htons(62078);
   226	    if (inet_pton(AF_INET, "10.7.0.1", &addr.sin_addr) != 1) {
   227	        [self log:@"[ERROR] inet_pton failed for 10.7.0.1."];
   228	        [self cleanupInternal];
   229	        return;
   230	    }
   231
   232	    [self log:@"[STEP 2] Calling idevice_tcp_provider_new..."];
   233	    err = idevice_tcp_provider_new((const idevice_sockaddr *)&addr, _pairingFile, "test-app", &_provider);
   234	    _pairingFile = NULL; // Consumed by the provider
   235	    if (err || !_provider) {
   236	        [self log:[NSString stringWithFormat:@"[ERROR] Provider creation: %s (%d)", err ? err->message : "NULL_ERR", err ? err->code : -1]];
   237	        if (err) { idevice_error_free(err); err = NULL; }
   238	        [self cleanupInternal];
   239	        return;
   240	    }
   241	    [self log:[NSString stringWithFormat:@"[OK] Provider created (%p).", _provider]];
   242
   243	    if (!check()) return;
   244	    [self log:@"[STEP 3] Calling lockdownd_connect..."];
   245	    err = lockdownd_connect(_provider, &_lockdown);
   246	    if (err || !_lockdown) {
   247	        [self log:[NSString stringWithFormat:@"[ERROR] Lockdownd connect: %s (%d)", err ? err->message : "NULL_ERR", err ? err->code : -1]];
   248	        if (err) { idevice_error_free(err); err = NULL; }
   249	        [self cleanupInternal];
   250	        return;
   251	    }
   252	    [self log:[NSString stringWithFormat:@"[OK] Lockdownd connected (%p).", _lockdown]];
   253
   254	    if (!check()) return;
   255	    [self log:@"[STEP 4] Calling lockdownd_start_session (TLS Handshake)..."];
   256	    struct IdevicePairingFile *sessionPairingFile = NULL;
   257	    err = idevice_provider_get_pairing_file(_provider, &sessionPairingFile);
   258	    if (err || !sessionPairingFile) {
   259	        [self log:[NSString stringWithFormat:@"[ERROR] Failed to get pairing file from provider: %s (%d)", err ? err->message : "NULL_ERR", err ? err->code : -1]];
   260	        if (err) { idevice_error_free(err); err = NULL; }
   261	        [self cleanupInternal];
   262	        return;
   263	    }
   264
   265	    err = lockdownd_start_session(_lockdown, sessionPairingFile);
   266	    idevice_pairing_file_free(sessionPairingFile); // We are responsible for this one
   267
   268	    if (err) {
   269	        [self log:[NSString stringWithFormat:@"[ERROR] Session start: %s (%d)", err->message ? err->message : "N/A", err->code]];
   270	        idevice_error_free(err); err = NULL;
   271	        [self cleanupInternal];
   272	        return;
   273	    }
   274	    [self log:@"[OK] Session established (Encrypted phase active)."];
   275	    [self fetchDeviceInfo:token];
   276
   277	    if (!check()) return;
   278	    [self log:@"[STEP 5] Calling heartbeat_connect..."];
   279	    err = heartbeat_connect(_provider, &_heartbeat);
   280	    if (err || !_heartbeat) {
   281	        [self log:[NSString stringWithFormat:@"[WARN] Heartbeat link failed: %s (%d)", err ? err->message : "NULL_ERR", err ? err->code : -1]];
   282	        if (err) { idevice_error_free(err); err = NULL; }
   283	    } else {
   284	        [self log:[NSString stringWithFormat:@"[OK] Heartbeat active (%p). Scheduling timer.", _heartbeat]];
   285	        if (err) { idevice_error_free(err); err = NULL; }
   286	        dispatch_async(dispatch_get_main_queue(), ^{
   287	            if (check()) {
   288	                self.heartbeatTimer = [NSTimer scheduledTimerWithTimeInterval:10.0 target:self selector:@selector(onHeartbeatTimer) userInfo:@(token) repeats:YES];
   289	            }
   290	        });
   291	    }
   292
   293	    if (!check()) return;
   294	    [self log:@"[STEP 6] Enabling Device persistence..."];
   295	    if (!check()) return;
   296	    [self log:@"[STEP 7] Calling installation_proxy_connect..."];
   297	    err = installation_proxy_connect(_provider, &_instproxy);
   298	    if (err || !_instproxy) {
   299	        [self log:[NSString stringWithFormat:@"[WARN] InstProxy connect failed: %s (%d)", err ? err->message : "NULL_ERR", err ? err->code : -1]];
   300	        if (err) { idevice_error_free(err); err = NULL; }
   301	    } else {
   302	        [self log:[NSString stringWithFormat:@"[OK] InstProxy connected (%p).", _instproxy]];
   303	        if (err) { idevice_error_free(err); err = NULL; }
   304	    }
   305	    if (!check()) return;
   306	    [self log:@"[STEP 8] Calling springboard_services_connect..."];
   307	    err = springboard_services_connect(_provider, &_springboard);
   308	    if (err || !_springboard) {
   309	        [self log:[NSString stringWithFormat:@"[WARN] Springboard connect failed: %s (%d)", err ? err->message : "NULL_ERR", err ? err->code : -1]];
   310	        if (err) { idevice_error_free(err); err = NULL; }
   311	    } else {
   312	        [self log:[NSString stringWithFormat:@"[OK] Springboard connected (%p).", _springboard]];
   313	        if (err) { idevice_error_free(err); err = NULL; }
   314	    }
   315	    dispatch_async(dispatch_get_main_queue(), ^{
   316	        if (check()) {
   317	            self.keepAliveTimer = [NSTimer scheduledTimerWithTimeInterval:30.0 target:self selector:@selector(onKeepAliveTimer) userInfo:@(token) repeats:YES];
   318	            [[UIApplication sharedApplication] setIdleTimerDisabled:YES];
   319	        }
   320	    });
   321
   322	    [self updateStatus:@"Connected" color:[UIColor systemGreenColor]];
   323	    [self log:@"[SUCCESS] Full connection sequence completed."];
   324	}

   326	- (void)onHeartbeatTimer {
   327	    NSInteger token = [[self.heartbeatTimer userInfo] integerValue];
   328	    dispatch_async(_connectionQueue, ^{
   329	        if (self->_activeToken != token || !self->_heartbeat) return;
   330	        [self log:@"[HB] Sending Polo..."];
   331	        struct IdeviceFfiError *err = heartbeat_send_polo(self->_heartbeat);
   332	        if (!err) {
   333	            uint64_t interval = 0;
   334	            err = heartbeat_get_marco(self->_heartbeat, 1000, &interval);
   335	            if (!err) [self log:[NSString stringWithFormat:@"[HB] Marco received. Interval: %llu", interval]];
   336	        }
   337	        if (err) {
   338	            [self log:[NSString stringWithFormat:@"[HB] FAILED: %s (%d)", err->message ? err->message : "N/A", err->code]];
   339	            idevice_error_free(err); err = NULL;
   340	        }
   341	    });
   342	}

   344	- (void)onKeepAliveTimer {
   345	    NSInteger token = [[self.keepAliveTimer userInfo] integerValue];
   346	    dispatch_async(_connectionQueue, ^{
   347	        if (self->_activeToken != token || !self->_lockdown) return;
   348	        [self log:@"[KA] Querying DeviceName..."];
   349	        plist_t val = NULL;
   350	        struct IdeviceFfiError *err = lockdownd_get_value(self->_lockdown, "DeviceName", NULL, &val);
   351	        if (!err && val) {
   352	            char *name = NULL;
   353	            plist_get_string_val(val, &name);
   354	            if (name) { [self log:[NSString stringWithFormat:@"[KA] Verified Name: %s", name]]; plist_mem_free(name); }
   355	            plist_free(val);
   356	        } else {
   357	            [self log:[NSString stringWithFormat:@"[KA] FAILED: %s (%d)", err ? err->message : "NULL_VAL", err ? err->code : -1]];
   358	            if (err) { idevice_error_free(err); err = NULL; }
   359	            [self cleanupInternal];
   360	        }
   361	    });
   362	}
   363	- (void)fetchDeviceInfo:(NSInteger)token {
   364	    dispatch_async(_connectionQueue, ^{
   365	        if (self->_activeToken != token || !self->_lockdown) return;
   366	        [self log:@"[INFO] Fetching detailed device information..."];
   367
   368	        NSArray *keys = @[@"DeviceName", @"ProductVersion", @"ProductType", @"UniqueDeviceID", @"SerialNumber", @"CPUArchitecture", @"DeviceClass"];
   369	        NSMutableString *infoSummary = [NSMutableString stringWithString:@"\n[DEVICE INFO]\n"];
   370
   371	        for (NSString *key in keys) {
   372	            plist_t val = NULL;
   373	            struct IdeviceFfiError *err = lockdownd_get_value(self->_lockdown, [key UTF8String], NULL, &val);
   374	            if (!err && val) {
   375	                id obj = [self objectFromPlist:val];
   376	                if (obj) {
   377	                    [infoSummary appendFormat:@"  %-16s: %@\n", [key UTF8String], obj];
   378	                }
   379	                plist_free(val);
   380	            } else {
   381	                if (err) idevice_error_free(err);
   382	                [infoSummary appendFormat:@"  %-16s: [ERROR]\n", [key UTF8String]];
   383	            }
   384	        }
   385	        [infoSummary appendString:@"----------------"];
   386	        [self log:infoSummary];
   387	    });

   404	- (void)fetchAppList:(NSInteger)token {
   405	    dispatch_async(_connectionQueue, ^{
   406	        if (self->_activeToken != token || !self->_instproxy) return;
   407	        [self log:@"[APPS] Fetching app list..."];
   408
   409	        void *result = NULL;
   410	        size_t len = 0;
   411	        struct IdeviceFfiError *err = installation_proxy_get_apps(self->_instproxy, NULL, NULL, 0, &result, &len);
   412
   413	        if (err || !result) {
   414	            [self log:[NSString stringWithFormat:@"[ERROR] Failed to get apps: %s (%d)", err ? err->message : "NULL_RES", err ? err->code : -1]];
   415	            if (err) idevice_error_free(err);
   416	            return;
   417	        }
   418
   419	        NSMutableArray *apps = [NSMutableArray array];
   420	        plist_t *plistArray = (plist_t *)result;
   421
   422	        for (size_t i = 0; i < len; i++) {
   423	            plist_t appEntry = plistArray[i];
   424	            id obj = [self objectFromPlist:appEntry];
   425	            if ([obj isKindOfClass:[NSDictionary class]]) {
   426	                [apps addObject:obj];
   427	            }
   428	        }
   429	        idevice_plist_array_free(plistArray, len);
   430
   431	        [apps sortUsingComparator:^NSComparisonResult(NSDictionary *obj1, NSDictionary *obj2) {
   432	            NSString *name1 = obj1[@"CFBundleDisplayName"] ?: obj1[@"CFBundleName"] ?: @"";
   433	            NSString *name2 = obj2[@"CFBundleDisplayName"] ?: obj2[@"CFBundleName"] ?: @"";
   434	            return [name1 localizedCaseInsensitiveCompare:name2];
   435	        }];
   436
   437	        dispatch_async(dispatch_get_main_queue(), ^{
   438	            if (self->_activeToken == token) {
   439	                self.appList = [apps copy];
   440	                [self.tableView reloadData];
   441	                [self log:[NSString stringWithFormat:@"[APPS] Successfully loaded %lu apps.", (unsigned long)apps.count]];
   442	            }
   443	        });
   444	    });
   445	}

   496	    }
   497	}
   498
   499	        NSString *key = self.detailSections[indexPath.section][indexPath.row];

   563	- (id)objectFromPlist:(plist_t)plist {
   564	    if (!plist) return nil;
   565	    char *xml = NULL;
   566	    uint32_t len = 0;
   567	    if (plist_to_xml(plist, &xml, &len) != PLIST_ERR_SUCCESS || !xml) return nil;
   568
   569	    NSData *data = [NSData dataWithBytesNoCopy:xml length:len freeWhenDone:NO];
   570	    id obj = [NSPropertyListSerialization propertyListWithData:data options:NSPropertyListImmutable format:NULL error:NULL];
   571
   572	    plist_mem_free(xml);
   573	    return obj;
   574	}

   576	- (void)fetchIconForBundleId:(NSString *)bundleId completion:(void (^)(UIImage *))completion {
   577	    dispatch_async(_connectionQueue, ^{
   578	        if (!self->_springboard) {
   579	            completion([UIImage systemImageNamed:@"app.dashed"]);
   580	            return;
   581	        }
   582
   583	        void *data = NULL;
   584	        size_t len = 0;
   585	        struct IdeviceFfiError *err = springboard_services_get_icon(self->_springboard, [bundleId UTF8String], &data, &len);
   586
   587	        UIImage *icon = nil;
   588	        if (!err && data && len > 0) {
   589	            NSData *pngData = [NSData dataWithBytes:data length:len];
   590	            icon = [UIImage imageWithData:pngData];
   591	            idevice_data_free((uint8_t *)data, (uintptr_t)len);
   592	        } else {
   593	            if (err) idevice_error_free(err);
   594	            icon = [UIImage systemImageNamed:@"app.dashed"];
   595	        }
   596
   597	        if (icon) [self.iconCache setObject:icon forKey:bundleId];
   598	        completion(icon ?: [UIImage systemImageNamed:@"app.dashed"]);
   599	    });
   600	}
