   464	- (UITableViewCell *)tableView:(UITableView *)tableView cellForRowAtIndexPath:(NSIndexPath *)indexPath {
   465	    if (tableView == self.tableView) {
   466	        UITableViewCell *cell = [tableView dequeueReusableCellWithIdentifier:@"AppCell"];
   467	        if (!cell) {
   468	            cell = [[UITableViewCell alloc] initWithStyle:UITableViewCellStyleSubtitle reuseIdentifier:@"AppCell"];
   469	            cell.backgroundColor = [UIColor clearColor];
   470	            cell.textLabel.font = [UIFont boldSystemFontOfSize:14];
   471	            cell.detailTextLabel.font = [UIFont systemFontOfSize:12];
   472	            cell.accessoryType = UITableViewCellAccessoryDisclosureIndicator;
   473	        }
   474
   475	        NSDictionary *app = self.appList[indexPath.row];
   476	        cell.textLabel.text = app[@"CFBundleDisplayName"] ?: app[@"CFBundleName"] ?: @"Unknown";
   477	        cell.detailTextLabel.text = app[@"CFBundleIdentifier"];
   478	        NSString *bundleId = app[@"CFBundleIdentifier"];
   479	        UIImage *cachedIcon = [self.iconCache objectForKey:bundleId];
   480	        if (cachedIcon) {
   481	            cell.imageView.image = cachedIcon;
   482	        } else {
   483	            cell.imageView.image = [UIImage systemImageNamed:@"app.dashed"];
   484	            [self fetchIconForBundleId:bundleId completion:^(UIImage *icon) {
   485	                dispatch_async(dispatch_get_main_queue(), ^{
   486	                    UITableViewCell *updateCell = [tableView cellForRowAtIndexPath:indexPath];
   487	                    if (updateCell) {
   488	                        updateCell.imageView.image = icon;
   489	                        [updateCell setNeedsLayout];
   490	                    }
   491	    } else {
   492	        UITableViewCell *cell = [tableView dequeueReusableCellWithIdentifier:@"DetailCell"];
   493	        if (!cell) {
   494	            cell = [[UITableViewCell alloc] initWithStyle:UITableViewCellStyleValue1 reuseIdentifier:@"DetailCell"];
   495	            cell.backgroundColor = [UIColor secondarySystemGroupedBackgroundColor];
   496	            cell.textLabel.font = [UIFont boldSystemFontOfSize:12];
   497	            cell.detailTextLabel.font = [UIFont fontWithName:@"Menlo" size:12] ?: [UIFont monospacedSystemFontOfSize:12 weight:UIFontWeightRegular];
   498	            cell.detailTextLabel.numberOfLines = 0;
   499	        }
   500
   501	        NSString *key = self.detailSections[indexPath.section][indexPath.row];
   502	        cell.textLabel.text = key;
   503	        cell.detailTextLabel.text = [NSString stringWithFormat:@"%@", self.selectedAppDetails[key] ?: @"N/A"];
   504	        return cell;
   505	    }
   506	                });
   507	            }];
   508	        }
   509
   510	    return cell;
   511	}
